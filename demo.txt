# ---------- 3.1 Create repo, add README, commit, push ----------

git init
echo "# My Project" > README.md
git add README.md
git commit -m "Initial commit"

git remote add origin https://github.com/your-username/repo.git
git branch -M main
git push -u origin main


# ---------- 3.2 Create branch MITS, modify, commit, merge ----------

git branch MITS
git checkout MITS

echo "Change from MITS branch" >> README.md
git add .
git commit -m "Updated in MITS branch"

git checkout main
git merge MITS


# ---------- 3.3 View commit history and hash ----------

git log


# ---------- 3.4 Clone repository ----------

git clone https://github.com/your-username/repo.git


# ---------- 3.5 Push changes to remote ----------

echo "Another change" >> README.md
git add .
git commit -m "Another update"

# ---------- PROGRAM 1: Open Google ----------

from selenium import webdriver
from selenium.webdriver.common.by import By
import time
driver = webdriver.Edge()
driver.get("https://www.google.com")
time.sleep(5)
e = driver.find_element(By.NAME, "q")
print(e.tag_name, e.get_attribute("type"))
driver.quit()


# ---------- PROGRAM 2: Search in Google ----------


from selenium import webdriver
from selenium.webdriver.common.by import By
driver = webdriver.Edge()
driver.get("https://www.google.com")
search = driver.find_element(By.NAME, "q")
btn = driver.find_element(By.NAME, "btnK")
print(search.is_displayed(), btn.is_enabled())
search.send_keys("Selenium")
driver.quit()


# ---------- PROGRAM 3: Login Example ----------


from selenium import webdriver
from selenium.webdriver.common.by import By
driver = webdriver.Edge()
driver.get("https://www.facebook.com")
